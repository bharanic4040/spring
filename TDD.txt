package test;

import static org.junit.Assert.*;

import org.junit.Test;

import main.Date;
import main.DateUtils;

public class DateUtilsTest {

	@Test
	public void testSameMonthYearDay() {
		Date fromDate=new Date(12,3,2018);
		Date toDate=new Date(12,3,2018);
		assertEquals(0,DateUtils.duration(fromDate,toDate));
	}
	
	@Test
	public void testSameMonthAndYearDiffDay() {
		Date fromDate=new Date(12,3,2018);
		Date toDate=new Date(14,3,2018);
		assertEquals(2,DateUtils.duration(fromDate,toDate));
	}
	@Test
	public void testSameYearDiffDayAndMonth() {
		Date fromDate=new Date(12,3,2018);
		Date toDate=new Date(13,4,2018);
		assertEquals(32,DateUtils.duration(fromDate,toDate));
	}
	@Test
	public void testDiffYearDayAndMonth() {
		Date fromDate=new Date(12,3,2018);
		Date toDate=new Date(13,4,2019);
		assertEquals(366,DateUtils.duration(fromDate,toDate));
	}
	@Test
	public void testLeapYear() {
		Date fromDate=new Date(12,1,2016);
		Date toDate=new Date(12,1,2017);
		assertEquals(366,DateUtils.duration(fromDate,toDate));
	}
	@Test
	public void testNonLeapYear() {
		Date fromDate=new Date(12,1,2017);
		Date toDate=new Date(12,1,2018);
		assertEquals(365,DateUtils.duration(fromDate,toDate));
	}
	
	@Test
	public void testFebMonthLeapYear() {
		Date fromDate=new Date(1,2,2012);
		Date toDate=new Date(1,3,2012);
		assertEquals(29,DateUtils.duration(fromDate,toDate));
	}
	@Test
	public void testFebMonthNonLeapYear() {
		Date fromDate=new Date(1,2,2017);
		Date toDate=new Date(1,3,2017);
		assertEquals(28,DateUtils.duration(fromDate,toDate));
	}

}
